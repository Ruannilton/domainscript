func TestDecimalAddPreservesExactPrecision(t *testing.T) {
	a, err := ParseDecimal("0.1")
	if err != nil {
		t.Fatal(err)
	}
	b, err := ParseDecimal("0.2")
	if err != nil {
		t.Fatal(err)
	}
	got := a.Add(b)
	want, err := ParseDecimal("0.3")
	if err != nil {
		t.Fatal(err)
	}
	// Em float64, 0.1+0.2 é 0.30000000000000004: aqui tem que bater exato.
	if got.Cmp(want) != 0 {
		t.Fatalf("Add impreciso: got %s, want %s", got, want)
	}
	if got.String() != "0.3000" {
		t.Fatalf("String() = %q, want %q", got.String(), "0.3000")
	}
}

func TestDecimalParseRoundsHalfEven(t *testing.T) {
	cases := []struct{ in, want string }{
		{"1.23445", "1.2344"},   // descarte exatamente .5 -> arredonda pro par (4 fica)
		{"1.23455", "1.2346"},   // descarte exatamente .5 -> arredonda pro par (5 -> 6)
		{"1.23459", "1.2346"},   // descarte .9 -> não é empate, arredonda pra cima
		{"1.23441", "1.2344"},   // descarte .1 -> não é empate, arredonda pra baixo
		{"-1.23445", "-1.2344"}, // sinal preservado, mesma regra de empate
	}
	for _, c := range cases {
		got, err := ParseDecimal(c.in)
		if err != nil {
			t.Fatalf("ParseDecimal(%q): %v", c.in, err)
		}
		if got.String() != c.want {
			t.Fatalf("ParseDecimal(%q).String() = %q, want %q", c.in, got.String(), c.want)
		}
	}
}

func TestMemoryEventStoreAppendLoadPreservesOrder(t *testing.T) {
	store := NewMemoryEventStore()
	ctx := context.Background()

	first := &testEvent{Msg: "first"}
	second := &testEvent{Msg: "second"}
	if err := store.Append(ctx, "wallet-1", []Event{first}); err != nil {
		t.Fatal(err)
	}
	if err := store.Append(ctx, "wallet-1", []Event{second}); err != nil {
		t.Fatal(err)
	}

	events, err := store.Load(ctx, "wallet-1")
	if err != nil {
		t.Fatal(err)
	}
	if len(events) != 2 {
		t.Fatalf("esperava 2 eventos, got %d", len(events))
	}
	got1 := events[0].(*testEvent)
	got2 := events[1].(*testEvent)
	if got1.Msg != "first" || got2.Msg != "second" {
		t.Fatalf("ordem não preservada: %q, %q", got1.Msg, got2.Msg)
	}
	if got1.Sequence != 1 || got2.Sequence != 2 {
		t.Fatalf("Sequence não incremental: %d, %d", got1.Sequence, got2.Sequence)
	}
	if got1.AggregateID != "wallet-1" || got2.AggregateID != "wallet-1" {
		t.Fatalf("AggregateID não atribuído: %q, %q", got1.AggregateID, got2.AggregateID)
	}
	if got1.Timestamp.IsZero() {
		t.Fatal("Timestamp não atribuído")
	}

	// Um aggregateId desconhecido devolve stream vazia, sem erro.
	empty, err := store.Load(ctx, "desconhecido")
	if err != nil || len(empty) != 0 {
		t.Fatalf("Load de id desconhecido: events=%v err=%v", empty, err)
	}
}

func TestUnitOfWorkTxLoadReturnsAppendedEvents(t *testing.T) {
	store := NewMemoryEventStore()
	uow := NewUnitOfWork(store)

	if err := uow.Run(context.Background(), func(tx Tx) error {
		return tx.Append("wallet-1", []Event{&testEvent{Msg: "first"}})
	}); err != nil {
		t.Fatal(err)
	}

	// Tx.Load é lido numa unit of work SEPARADA (o padrão real: LoadWallet
	// reconstrói o aggregate numa unit of work antes de um Handle mutar mais
	// eventos nela) — prova que a leitura enxerga o que uma unit of work
	// anterior já commitou no store durável.
	err := uow.Run(context.Background(), func(tx Tx) error {
		events, err := tx.Load("wallet-1")
		if err != nil {
			return err
		}
		if len(events) != 1 {
			t.Fatalf("esperava 1 evento via Tx.Load, got %d", len(events))
		}
		got := events[0].(*testEvent)
		if got.Msg != "first" {
			t.Fatalf("Msg incorreto: got %q", got.Msg)
		}
		return nil
	})
	if err != nil {
		t.Fatal(err)
	}

	// aggregateID desconhecido: stream vazia, sem erro (mesmo contrato de
	// EventStore.Load).
	err = uow.Run(context.Background(), func(tx Tx) error {
		events, err := tx.Load("desconhecido")
		if err != nil || len(events) != 0 {
			t.Fatalf("Tx.Load de id desconhecido: events=%v err=%v", events, err)
		}
		return nil
	})
	if err != nil {
		t.Fatal(err)
	}
}

func TestSemaphoreAcquireBlocksAtLimitThenReleaseFrees(t *testing.T) {
	sem := NewSemaphore(1)
	ctx := context.Background()
	if err := sem.Acquire(ctx); err != nil {
		t.Fatal(err)
	}

	acquired := make(chan struct{})
	go func() {
		if err := sem.Acquire(ctx); err != nil {
			return
		}
		close(acquired)
	}()

	select {
	case <-acquired:
		t.Fatal("Acquire deveria bloquear enquanto o único slot está em uso")
	case <-time.After(20 * time.Millisecond):
	}

	sem.Release()
	select {
	case <-acquired:
	case <-time.After(time.Second):
		t.Fatal("Acquire deveria destravar depois de Release")
	}
}

func TestRetryPolicyRunRespectsCtxCancellationDuringBackoff(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	policy := RetryPolicy{Attempts: 5, BaseDelay: time.Hour} // backoff bem maior que o teste pode esperar
	calls := 0
	done := make(chan error, 1)
	go func() {
		done <- policy.Run(ctx, func() error {
			calls++
			return errors.New("falha")
		})
	}()
	time.Sleep(10 * time.Millisecond)
	cancel()

	select {
	case err := <-done:
		if !errors.Is(err, context.Canceled) {
			t.Fatalf("esperava context.Canceled, got %v", err)
		}
	case <-time.After(time.Second):
		t.Fatal("Run deveria retornar assim que ctx é cancelado durante o backoff")
	}
	if calls != 1 {
		t.Fatalf("esperava 1 tentativa antes do cancelamento durante o backoff, got %d", calls)
	}
}

func TestCronScheduleNextFindsDailyTimeAtOrAfterFrom(t *testing.T) {
	sched, err := ParseCron("30 2 * * *") // 02:30 todo dia
	if err != nil {
		t.Fatal(err)
	}
	from := time.Date(2026, 7, 3, 10, 0, 0, 0, time.UTC) // depois das 02:30 de hoje
	next := sched.Next(from)

func TestQueueChannelConcurrencyBoundsSimultaneousDeliveries(t *testing.T) {
	const concurrency = 3
	// keys >> concurrency: com Concurrency shards fixos (ver a doc de
	// queueChannel, channel.go.txt) e chaves DISTINTAS hasheadas por
	// FNV-1a, muito mais chaves que shards torna praticamente certo que
	// todo shard recebe ao menos um item cedo (o que o poll abaixo espera
	// — não é uma suposição de timing, é aritmética de hashing: com só
	// `concurrency` chaves haveria risco real de um shard ficar ocioso).
	const keys = 60

	var inFlight int32
	var maxObserved int32
	release := make(chan struct{})
	var wg sync.WaitGroup
	wg.Add(keys)

	ch := NewQueueChannel(QueueChannelConfig{Concurrency: concurrency}, func(ev Event) string {
		return ev.(*testEvent).Msg // cada evento tem uma chave DISTINTA
	})
	ch.Subscribe("TestEvent", func(ctx context.Context, ev Event) error {
		n := atomic.AddInt32(&inFlight, 1)
		for {
			cur := atomic.LoadInt32(&maxObserved)
			if n <= cur || atomic.CompareAndSwapInt32(&maxObserved, cur, n) {
				break
			}
		}
		<-release // segura a entrega até o teste liberar, forçando sobreposição
		atomic.AddInt32(&inFlight, -1)
		wg.Done()
		return nil
	})

	// Publish roda em goroutine própria: com BatchSize default (1) cada
	// shard só cabe 1 item em voo (bloqueado em <-release) + 1 no buffer —
	// publicar as `keys` chaves de uma vez na goroutine do teste
	// AUTO-DEADLOCKARIA (Publish bloqueia no shard cheio, e só o teste,
	// depois de terminar de publicar, chega ao close(release) que libera
	// espaço). Rodar em background deixa o teste seguir para o poll de
	// inFlight abaixo enquanto o shard vai esvaziando (1 em voo + 1 em
	// buffer) conforme cada Publish consegue enfileirar.
	ctx := context.Background()
	publishDone := make(chan struct{})
	go func() {
		defer close(publishDone)
		for i := 0; i < keys; i++ {
			if err := ch.Publish(ctx, &testEvent{Msg: fmt.Sprintf("key-%d", i)}); err != nil {
				t.Errorf("Publish: %v", err)
				return
			}
		}
	}()

	// Espera inFlight estabilizar em exatamente `concurrency` (um shard
	// cada, o resto bloqueado atrás no buffer do próprio shard) — poll em
	// vez de um sleep fixo, robusto a máquinas mais lentas.
	deadline := time.Now().Add(2 * time.Second)
	for time.Now().Before(deadline) && atomic.LoadInt32(&inFlight) != concurrency {
		time.Sleep(5 * time.Millisecond)
	}
	if got := atomic.LoadInt32(&inFlight); got != concurrency {
		t.Fatalf("esperava exatamente %d entregas simultâneas em voo (Concurrency, um shard cada), got %d", concurrency, got)
	}
	close(release)

	waitDone := make(chan struct{})
	go func() { wg.Wait(); close(waitDone) }()
	select {
	case <-waitDone:
	case <-time.After(3 * time.Second):
		t.Fatal("timeout esperando as entregas restantes")
	}
	if got := atomic.LoadInt32(&maxObserved); got != concurrency {
		t.Fatalf("maxObserved = %d, want exatamente %d (Concurrency)", got, concurrency)
	}

	select {
	case <-publishDone:
	case <-time.After(3 * time.Second):
		t.Fatal("timeout esperando a goroutine de publish terminar")
	}
}

func TestQueueChannelPublishBlocksWhenPartitionSaturatedThenUnblocks(t *testing.T) {
	block := make(chan struct{})
	ch := NewQueueChannel(QueueChannelConfig{Concurrency: 1, BatchSize: 1}, func(ev Event) string {
		return "k" // uma única partição, cuja capacidade é exatamente BatchSize=1
	})
	ch.Subscribe("TestEvent", func(ctx context.Context, ev Event) error {
		<-block // segura o consumidor indefinidamente até o teste liberar
		return nil
	})

	ctx := context.Background()
	if err := ch.Publish(ctx, &testEvent{Msg: "1"}); err != nil {
		t.Fatal(err)
	}
	// dá tempo do drain pegar o item 1 e travar no handler, esvaziando o buffer.
	time.Sleep(50 * time.Millisecond)
	if err := ch.Publish(ctx, &testEvent{Msg: "2"}); err != nil {
		t.Fatal(err) // enche o buffer (capacidade 1)
	}

	publish3 := make(chan error, 1)
	go func() { publish3 <- ch.Publish(ctx, &testEvent{Msg: "3"}) }()

	select {
	case <-publish3:
		t.Fatal("Publish deveria bloquear com a partição saturada (BatchSize=1) e o consumidor preso no handler")
	case <-time.After(100 * time.Millisecond):
		// esperado: ainda bloqueado (backpressure).
	}

	close(block)
	select {
	case err := <-publish3:
		if err != nil {
			t.Fatal(err)
		}
	case <-time.After(2 * time.Second):
		t.Fatal("Publish deveria destravar assim que o consumidor for liberado")
	}
}

func TestQueueChannelCircuitBreakerSkipsDeliveryWhenOpen(t *testing.T) {
	var calls int32
	ch := NewQueueChannel(QueueChannelConfig{Concurrency: 1, BreakerThreshold: 2, BreakerCooldown: time.Hour}, func(ev Event) string {
		return "k" // uma única partição: entrega estritamente sequencial
	})
	sentinel := errors.New("falha")
	handled := make(chan struct{}, 10)
	ch.Subscribe("TestEvent", func(ctx context.Context, ev Event) error {
		atomic.AddInt32(&calls, 1)
		handled <- struct{}{}
		return sentinel
	})

	ctx := context.Background()
	// 2 falhas consecutivas abrem o circuito (threshold=2).
	for i := 0; i < 2; i++ {
		if err := ch.Publish(ctx, &testEvent{Msg: fmt.Sprint(i)}); err != nil {
			t.Fatal(err)
		}
		select {
		case <-handled:
		case <-time.After(time.Second):
			t.Fatalf("timeout esperando a entrega %d", i)
		}
	}

	// 3ª entrega: circuito já deveria estar aberto -> handler NÃO deveria rodar.
	if err := ch.Publish(ctx, &testEvent{Msg: "3"}); err != nil {
		t.Fatal(err)
	}
	select {
	case <-handled:
		t.Fatal("NÃO esperava o handler rodar com o circuito aberto")
	case <-time.After(200 * time.Millisecond):
		// esperado: nada chamado (a 3ª entrega foi pulada, ver deliver em
		// channel.go.txt).
	}
	if got := atomic.LoadInt32(&calls); got != 2 {
		t.Fatalf("calls = %d, want 2 (a 3ª entrega deveria ter sido pulada pelo circuito aberto)", got)
	}
}

	}

func TestQueueChannelRecoversFromPanickingHandlerAndKeepsProcessing(t *testing.T) {
	// Mesma chave para as duas entregas: prova que o shard (a goroutine de
	// drain) sobrevive ao panic da 1ª entrega e continua processando a 2ª
	// em vez de travar o shard inteiro (e, com ele, toda futura entrega de
	// qualquer chave que hasheie para o mesmo shard).
	ch := NewQueueChannel(QueueChannelConfig{Concurrency: 1}, func(ev Event) string {
		return "k"
	})
	processed := make(chan string, 1)
	ch.Subscribe("TestEvent", func(ctx context.Context, ev Event) error {
		msg := ev.(*testEvent).Msg
		if msg == "boom" {
			panic("handler explodiu")
		}
		processed <- msg
		return nil
	})

	ctx := context.Background()
	if err := ch.Publish(ctx, &testEvent{Msg: "boom"}); err != nil {
		t.Fatal(err)
	}
	if err := ch.Publish(ctx, &testEvent{Msg: "depois-do-panic"}); err != nil {
		t.Fatal(err)
	}

	select {
	case msg := <-processed:
		if msg != "depois-do-panic" {
			t.Fatalf("Msg = %q, want %q", msg, "depois-do-panic")
		}
	case <-time.After(time.Second):
		t.Fatal("esperava que a entrega seguinte ao panic ainda fosse processada (isolamento por entrega, o shard não deveria travar)")
	}
}

func TestQueueChannelPanicCountsAsCircuitBreakerFailure(t *testing.T) {
	ch := NewQueueChannel(QueueChannelConfig{Concurrency: 1, BreakerThreshold: 1, BreakerCooldown: time.Hour}, func(ev Event) string {
		return "k"
	})
	handled := make(chan struct{}, 10)
	var secondCallSeen int32
	ch.Subscribe("TestEvent", func(ctx context.Context, ev Event) error {
		defer func() { handled <- struct{}{} }()
		if ev.(*testEvent).Msg == "boom" {
			panic("handler explodiu")
		}
		atomic.StoreInt32(&secondCallSeen, 1)
		return nil
	})

	ctx := context.Background()
	if err := ch.Publish(ctx, &testEvent{Msg: "boom"}); err != nil {
		t.Fatal(err)
	}
	select {
	case <-handled:
	case <-time.After(time.Second):
		t.Fatal("timeout esperando a 1ª entrega (panic)")
	}

	// threshold=1: o panic sozinho já deveria ter aberto o circuito — a 2ª
	// entrega (que NÃO teria dado panic) deveria ser pulada pelo breaker,
	// então secondCallSeen nunca vira 1.
	if err := ch.Publish(ctx, &testEvent{Msg: "nunca deveria rodar"}); err != nil {
		t.Fatal(err)
	}
	select {
	case <-handled:
		t.Fatal("NÃO esperava o handler rodar: o circuito deveria estar aberto depois do panic")
	case <-time.After(200 * time.Millisecond):
		// esperado: nada chamado.
	}
	if atomic.LoadInt32(&secondCallSeen) != 0 {
		t.Fatal("NÃO esperava a 2ª entrega ser processada com o circuito aberto")
	}
}
