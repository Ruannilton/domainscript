package runtime

import "time"

// Event is implemented by every domain event a Handle emits.
type Event interface {
	// EventType returns the stable event name used for serialization,
	// registry lookup, and dispatch — independent of the Go type name.
	EventType() string
	// SetMeta replaces the event's embedded EventMeta. Every generated
	// event struct gets this for free by embedding EventMeta (see
	// EventMeta.SetMeta): EventStore.Append uses it to stamp
	// AggregateID/Sequence/Timestamp without reflection, which is why
	// events flow through the runtime as pointers (*GeneratedEvent) rather
	// than values — SetMeta has a pointer receiver so it promotes onto the
	// pointer type only.
	SetMeta(meta EventMeta)
}

// EventMeta is embedded by every generated event struct to carry the
// metadata implicit in the event-sourcing model (spec §4.2): which
// aggregate stream the event belongs to, its position in that stream, and
// when it was recorded. EventType is deliberately not a field here: it is
// the Event.EventType() method on the embedding type, so there is a single
// source of truth for it instead of two.
type EventMeta struct {
	// AggregateID is the id of the aggregate stream this event belongs to —
	// the key EventStore stores/loads by. It is metadata stamped on append,
	// not necessarily any "id" field on the event's own payload.
	AggregateID string `json:"aggregateId"`
	// Sequence is the 1-based position of this event within its aggregate
	// stream, assigned by EventStore.Append.
	Sequence int64 `json:"sequence"`
	// Timestamp is when EventStore.Append recorded the event, in UTC.
	Timestamp time.Time `json:"timestamp"`
}

// SetMeta replaces the metadata in place. It has a pointer receiver so it
// promotes to a pointer-to-event-struct method automatically wherever
// EventMeta is embedded by value — no per-event-type boilerplate needed.
func (m *EventMeta) SetMeta(meta EventMeta) { *m = meta }
