package runtime

// Caller represents the identity invoking a Handle or UseCase, used to
// evaluate `access` rules and by business logic that branches on identity.
// A concrete implementation is wired at the HTTP/gRPC edge (E9.2/H1); only
// the contract lives here.
type Caller interface {
	// Authenticated reports whether the request carries a verified identity.
	Authenticated() bool
	// ID returns the caller's stable identifier. Empty when not
	// Authenticated().
	ID() string
	// HasRole reports whether the caller holds role.
	HasRole(role string) bool
}

// testCaller is a minimal Caller stub used ONLY by generated tests (H4,
// *.test.ds, §22) that invoke a Handle/UseCase directly — the DS test
// grammar (given/when/then) has no syntax to express "as caller X", so
// generated tests always call through NewTestCaller (below) instead of a
// per-test type.
type testCaller struct {
	authenticated bool
	id            string
	roles         map[string]bool
}

func (c testCaller) Authenticated() bool      { return c.authenticated }
func (c testCaller) ID() string               { return c.id }
func (c testCaller) HasRole(role string) bool { return c.roles[role] }

// NewTestCaller builds an authenticated Caller with the given id and roles —
// used by generated *.test.ds tests to drive a Handle/UseCase's `access`
// rule (§22): every generated scenario calls a Handle as an authenticated
// caller whose id matches the target aggregate, satisfying both
// `caller.authenticated` and `caller.id == self.id` style rules uniformly,
// since given/when/then has no way to express access denial scenarios.
func NewTestCaller(id string, roles ...string) Caller {
	set := make(map[string]bool, len(roles))
	for _, r := range roles {
		set[r] = true
	}
	return testCaller{authenticated: true, id: id, roles: set}
}
