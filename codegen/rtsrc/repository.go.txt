package runtime

import (
	"context"
	"sync"
)

// Repository is the seam for StateStored aggregate persistence: it loads and
// saves the concrete state struct T directly, without event replay. T is the
// per-domain state struct generated for a specific Aggregate — unknown to
// this package, hence the generic.
//
// Multi-tenancy (G5, spec §13.2, row_level): the same tag-on-write/filter-
// on-read scheme as EventStore (see eventstore.go.txt's type doc) applies
// here — Save tags id's state with the Tenant.ID active in ctx (""
// otherwise), and Load reports found=false when the stored tag doesn't
// match ctx's tenant, exactly as if id had never been saved. A service that
// never puts a Tenant in ctx never tags anything, so this is a no-op there.
type Repository[T any] interface {
	// Load returns the stored state for id. found is false when no state
	// has been saved for id yet — or it was saved under a different tenant
	// than ctx's (§13.2), unless WithCrossTenantBypass is set — in which
	// case state is T's zero value.
	Load(ctx context.Context, id string) (state T, found bool, err error)
	// Save persists state under id, replacing any previously saved value.
	Save(ctx context.Context, id string, state T) error
}

// tenantState pairs a saved state value with the tenant ID it was first
// saved under — the Repository twin of tenantStream (eventstore.go.txt).
type tenantState[T any] struct {
	tenantID string
	state    T
}

// memoryRepository is a thread-safe, in-process Repository[T] backed by a
// map keyed by aggregate id.
type memoryRepository[T any] struct {
	mu     sync.RWMutex
	states map[string]*tenantState[T]
}

// NewMemoryRepository creates an empty in-memory Repository[T].
func NewMemoryRepository[T any]() Repository[T] {
	return &memoryRepository[T]{states: make(map[string]*tenantState[T])}
}

// Load implements Repository.
func (r *memoryRepository[T]) Load(ctx context.Context, id string) (T, bool, error) {
	r.mu.RLock()
	defer r.mu.RUnlock()
	ts := r.states[id]
	if ts == nil || !tenantVisible(ctx, ts.tenantID) {
		var zero T
		return zero, false, nil
	}
	return ts.state, true, nil
}

// Save implements Repository. The first Save for id stamps its tenant tag
// from ctx (§13.2); later Saves keep that original tag, mirroring
// memoryEventStore.Append.
func (r *memoryRepository[T]) Save(ctx context.Context, id string, state T) error {
	r.mu.Lock()
	defer r.mu.Unlock()
	ts := r.states[id]
	if ts == nil {
		ts = &tenantState[T]{tenantID: tenantIDOf(ctx)}
		r.states[id] = ts
	}
	ts.state = state
	return nil
}
