package runtime

import (
	"context"
	"encoding/json"
	"errors"
	"log/slog"
	"sync"
	"time"
)

// Outbox is the delivery seam for Policies declared with "delivery
// AtLeastOnce" (Marco F1, REQ-23.1) — as opposed to Dispatcher, which is
// explicitly documented as in-process/best-effort (see dispatcher.go.txt).
// Subscribe has the exact same shape as Dispatcher.Subscribe on purpose: a
// generated Policy handler registers on whichever seam its `delivery`
// clause calls for without changing its own signature.
//
// Marco F5 shipped memoryOutbox, the in-process stopgap below. Marco J
// (REQ-42.2/42.3, §design infra-providers 3.2) adds DurableOutbox, a real
// durable/retried implementation behind this SAME interface — generated
// code that calls Outbox.Subscribe does not change when a module switches
// from memoryOutbox to DurableOutbox (task J2.5 decides which one the
// generated wiring constructs).
type Outbox interface {
	// Subscribe registers handler to run for every event whose EventType()
	// equals eventType, with the AtLeastOnce delivery guarantee (durable and
	// retried for DurableOutbox; in-process/best-effort for memoryOutbox —
	// see each type's doc).
	Subscribe(eventType string, handler func(ctx context.Context, ev Event) error)
}

// memoryOutbox is the Marco F1 stopgap Outbox: it only forwards Subscribe to
// the Dispatcher it wraps. It is NOT durable yet — a process crash between
// Publish and handler execution loses the delivery, exactly like Dispatcher.
// This exists so AtLeastOnce Policies are registrable and their generated
// code compiles ahead of a real implementation, without pretending to a
// durability guarantee the runtime doesn't have yet (documented
// deliberately, see the interface doc above). Selected by codegen (task
// J2.5) for any module WITHOUT a real Database — DurableOutbox is opt-in.
type memoryOutbox struct {
	d Dispatcher
}

// NewOutbox creates the Marco F1 Outbox over d. Every AtLeastOnce Policy
// subscribes through this seam instead of Dispatcher directly, so swapping
// in a real durable implementation later is a change to this
// constructor/type only — no generated code changes.
func NewOutbox(d Dispatcher) Outbox {
	return &memoryOutbox{d: d}
}

// Subscribe implements Outbox.
func (o *memoryOutbox) Subscribe(eventType string, handler func(ctx context.Context, ev Event) error) {
	o.d.Subscribe(eventType, handler)
}

// --- DurableOutbox (Marco J, REQ-42.2/42.3, §design infra-providers 3.2) ---

// EventFactory constructs a new zero-value pointer of a concrete Event
// type, ready for json.Unmarshal — the SAME shape sqlruntime.EventFactory
// already uses (codegen/sqlrt/eventstore.go.txt, an alias too), so a
// module's generated "EventRegistry()" (map[string]func() Event) passes
// directly to NewDurableOutbox with zero conversion, same convention
// NewEventStore/NewUnitOfWork already established for the SQL adapter.
type EventFactory = func() Event

// OutboxRow is one pending (or previously-failed) delivery scanned from
// durable storage — the seam-level shape DurableOutbox works with,
// independent of which SQL adapter backs it. Payload is the raw encoded
// form (JSON, the same convention EventStore already uses); DurableOutbox
// decodes it into a concrete Event via the registry it was constructed
// with, so OutboxStore implementations never need to know about event
// types.
type OutboxRow struct {
	ID        int64
	EventType string
	Payload   []byte
	Attempts  int
}

// OutboxStore is the durable storage seam DurableOutbox polls (REQ-42.2/
// 42.3): implemented by the SQL adapter (sqlruntime.NewOutboxStore,
// codegen/sqlrt/outbox.go.txt) over the "outbox" table Tx.EnqueueOutbox
// already writes to (task J2.1) — DurableOutbox itself never imports
// database/sql or knows about a specific Dialect (NFR-12/21: the core stays
// free of any driver).
type OutboxStore interface {
	// ProcessBatch scans up to batch undelivered rows (FIFO, row-locked for
	// the whole batch so concurrent relay replicas never process the same
	// row twice — see Dialect.ScanUndelivered, codegen/sqlrt, for how the
	// lock is taken) and calls deliver once per row, INSIDE the same
	// transaction that holds the lock: deliver returning nil marks that row
	// delivered; a non-nil error leaves it undelivered with its attempts
	// counter incremented, to be re-scanned on a later ProcessBatch call
	// (at-least-once, REQ-42.2). Commits once every row in the batch has
	// been processed — a crash mid-batch rolls back the WHOLE batch (every
	// row stays exactly as it was before this call), never a partial mark.
	ProcessBatch(ctx context.Context, batch int, deliver func(row OutboxRow) error) (processed int, err error)

	// PurgeDelivered deletes every row whose delivered_at is non-NULL and
	// older than cutoff (task J2.5, REQ-42.7, §design infra-providers "boas
	// práticas de limpeza/retenção") — without this, the outbox table grows
	// without bound (every delivered row stays forever otherwise). Returns
	// the number of rows deleted. See Dialect.PurgeDelivered (codegen/sqlrt)
	// for the per-database SQL text.
	PurgeDelivered(ctx context.Context, cutoff time.Time) (purged int, err error)
}

// defaultOutboxPollInterval/defaultOutboxBatchSize are DurableOutbox's
// unconfigured defaults — same order of magnitude as
// idempotencyCleanupInterval (usecase_idempotency.go, a generated periodic
// worker in the same spirit): frequent enough that AtLeastOnce delivery
// feels near-real-time, infrequent enough not to hammer the database with
// an empty-outbox poll every tick. defaultOutboxMaxBackoff caps Start's
// exponential backoff on repeated ProcessBatch errors (revisão da PR #20):
// a database outage or a downstream dependency being down must not keep the
// relay hammering it once a second, flooding logs — see Start below.
const (
	defaultOutboxPollInterval = time.Second
	defaultOutboxBatchSize    = 50
	defaultOutboxMaxBackoff   = 10 * time.Second
)

// DurableOutbox is the Marco J real Outbox (REQ-42.2/42.3): Subscribe
// registers handlers exactly like memoryOutbox, but delivery is driven by
// Start's relay loop polling OutboxStore instead of an in-process
// Dispatcher — a row enqueued by Tx.EnqueueOutbox (atomically with the
// business transaction, task J2.1) survives a crash between commit and
// delivery, because the row is still sitting in durable storage, undelivered,
// for the NEXT ProcessBatch call to pick up.
//
// --- Composição outbox <-> canal (task J2.4, REQ-42.6, §design
// infra-providers 3.2a) ---
//
// publisher (optional, NewDurableOutbox's variadic last argument — same
// convention as NewUnitOfWork) is the module's ONE outgoing transport for
// cross-service delivery (a ChannelTransport, e.g. the RabbitMQ adapter once
// Marco J3 lands, or today's in-memory QueueChannel — both already satisfy
// Publisher structurally, same shape as Dispatcher). A single DurableOutbox
// routes ALL of its delivered rows ONE way, decided once at construction —
// never per-event_type dynamically — mirroring the SAME mutual-exclusivity
// codegen.go's producer-channel wiring already enforces for the (older,
// non-durable) UnitOfWork publisher: a module either has an outgoing
// channel (every emitted PublicEvent leaves the process through it) or
// Policies subscribe locally (in-process) — never a per-event mix within
// the SAME module, today.
//
//   - publisher != nil (cross-service): deliver calls publisher.Publish
//     instead of running any locally-subscribed handler — a
//     publish failure is retriable exactly like a handler failure (ProcessBatch
//     increments attempts instead of marking delivered, so the SAME row
//     is retried on the next Tick — REQ-42.6's durability claim: a crash
//     between commit and a successful Publish never loses the event,
//     because the row is still sitting undelivered in the outbox table for
//     Start's next tick to pick up).
//   - publisher == nil (in-process, the J2.3 shape): deliver runs every
//     handler Subscribe registered for that row's event type, exactly as
//     before this task.
//
// Wiring which module gets which (deciding when a module's UnitOfWork
// publisher STOPS receiving the channel directly, so an event is never
// published twice — once inline at commit, once again later by the relay)
// is task J2.5's job (codegen.go/cmd/<service>/main.go), not this type's:
// DurableOutbox itself just does whichever ONE thing it was constructed to
// do.
//
// --- Seleção/wiring gerado (task J2.5, REQ-42.5/42.7) ---
//
// codegen/decl_policy.go constructs a DurableOutbox, instead of the
// memoryOutbox stopgap above, only for a module whose Policy AtLeastOnce
// subscribes on "d" (the local Dispatcher path, not a cross-service "queue"
// channel — that case is unchanged, out of this task's scope) AND that owns
// a real Database (recognizedSQLProvider, codegen/sql_wiring.go) — a module
// without one keeps the memoryOutbox stopgap, so its generated wiring stays
// byte-identical (NFR-21/23). The generated cmd/<service>/main.go opens that
// Database's connection, builds a sqlruntime.OutboxStore over it, and calls
// two generated entry points on the module package: StartOutboxRelay(ctx)
// (this type's Start, above) and StartOutboxCleanup(ctx) (this type's
// Cleanup, below, on a periodic ticker — mirrors StartIdempotencyCleanup,
// usecase_idempotency.go).
type DurableOutbox struct {
	store     OutboxStore
	registry  map[string]EventFactory
	publisher Publisher

	mu       sync.Mutex
	handlers map[string][]func(ctx context.Context, ev Event) error
}

// NewDurableOutbox creates a DurableOutbox over store, decoding rows via
// registry (a module's generated "EventRegistry()", or a merge of several
// when a service consumes events declared by more than one module — see
// task J2.4/contracts.EventRegistry() for the cross-service case). publisher
// is optional (NewUnitOfWork's exact convention): when given, EVERY
// delivered row routes through it instead of locally-subscribed handlers —
// see the type doc above for the full routing rule.
func NewDurableOutbox(store OutboxStore, registry map[string]EventFactory, publisher ...Publisher) *DurableOutbox {
	o := &DurableOutbox{
		store:    store,
		registry: registry,
		handlers: make(map[string][]func(ctx context.Context, ev Event) error),
	}
	if len(publisher) > 0 {
		o.publisher = publisher[0]
	}
	return o
}

// Subscribe implements Outbox: registers handler for eventType. Multiple
// handlers for the SAME eventType (unusual but not forbidden) all run for
// every delivered row of that type — the row is marked delivered only if
// ALL of them succeed (mirrors memoryOutbox/Dispatcher's "all subscribers
// see every event" semantics).
func (o *DurableOutbox) Subscribe(eventType string, handler func(ctx context.Context, ev Event) error) {
	o.mu.Lock()
	defer o.mu.Unlock()
	o.handlers[eventType] = append(o.handlers[eventType], handler)
}

// Start runs the relay loop until ctx is done (the same "ticker + select"
// shape as a generated periodic worker, e.g. StartIdempotencyCleanup,
// usecase_idempotency.go) — called once from the generated
// cmd/<service>/main.go (task J2.5), alongside StartWorkers/Wire.
//
// Exponential backoff (revisão da PR #20): a ProcessBatch error (ex. o
// banco caiu, uma dependência downstream está fora) doubles the wait until
// defaultOutboxMaxBackoff instead of retrying every defaultOutboxPollInterval
// — a fixed 1s retry during an outage would hammer the database and flood
// logs with the same error every tick. A successful Tick (even one that
// simply found nothing to deliver) resets the interval back to the default.
func (o *DurableOutbox) Start(ctx context.Context) {
	interval := defaultOutboxPollInterval
	timer := time.NewTimer(interval)
	defer timer.Stop()
	for {
		select {
		case <-ctx.Done():
			return
		case <-timer.C:
			if _, err := o.Tick(ctx); err != nil {
				slog.Error("durable outbox: falha ao processar lote", "error", err)
				interval *= 2
				if interval > defaultOutboxMaxBackoff {
					interval = defaultOutboxMaxBackoff
				}
			} else {
				interval = defaultOutboxPollInterval
			}
			timer.Reset(interval)
		}
	}
}

// Tick processes one batch (defaultOutboxBatchSize rows) — exported so
// tests (and any custom scheduler) can drive delivery directly without
// waiting on Start's ticker.
func (o *DurableOutbox) Tick(ctx context.Context) (delivered int, err error) {
	return o.store.ProcessBatch(ctx, defaultOutboxBatchSize, func(row OutboxRow) error {
		return o.deliver(ctx, row)
	})
}

// Cleanup deletes every delivered row older than retention (task J2.5,
// REQ-42.7) — delegates to OutboxStore.PurgeDelivered, which does the actual
// per-database DELETE (Dialect.PurgeDelivered, codegen/sqlrt). Called
// periodically by the generated StartOutboxCleanup(ctx), never inline in the
// relay loop (Start above): cleanup and delivery are independent concerns
// with independent cadences, the same reasoning that keeps
// StartIdempotencyCleanup (usecase_idempotency.go) a separate worker from
// the request path it cleans up after.
func (o *DurableOutbox) Cleanup(ctx context.Context, retention time.Duration) (purged int, err error) {
	// o.store == nil só acontece se NewDurableOutbox foi chamado com store
	// nil diretamente (revisão da PR #22) — um erro claro em vez de um nil
	// pointer panic dentro do worker periódico gerado (StartOutboxCleanup).
	if o.store == nil {
		return 0, errors.New("runtime: DurableOutbox.Cleanup: store nil (NewDurableOutbox foi chamado sem um OutboxStore)")
	}
	return o.store.PurgeDelivered(ctx, time.Now().Add(-retention))
}

// deliver decodes row via the registry, then routes it to EXACTLY ONE
// destination (task J2.4, REQ-42.6, §design infra-providers 3.2a — see the
// type doc's "Composição outbox <-> canal" section for the full rule):
// o.publisher when configured (cross-service), or every locally-subscribed
// handler for its event type otherwise (in-process, the J2.3 shape). An
// unknown event type or a JSON decode failure is a permanent, non-retriable
// condition (no future ProcessBatch call would decode it any differently) —
// logged loudly and treated as delivered (marked, not retried) so a single
// malformed/unrecognized row can never wedge the relay in a tight retry
// loop forever (the poison-pill concern §design infra-providers 3.3 raises
// for the RabbitMQ transport applies here too, in spirit). A publish or
// handler failure, in contrast, IS retriable (transient — a downstream
// dependency being briefly unavailable, say), so it propagates to
// ProcessBatch, which increments attempts instead of marking delivered —
// the SAME row is re-scanned (ORDER BY attempts ASC, id ASC still puts it
// ahead of never-yet-tried rows) and re-delivered on a later Tick, closing
// exactly the "crash between commit and publish" window REQ-42.6 exists
// for.
func (o *DurableOutbox) deliver(ctx context.Context, row OutboxRow) error {
	factory, ok := o.registry[row.EventType]
	if !ok {
		slog.Error("durable outbox: tipo de evento desconhecido, descartando linha", "eventType", row.EventType, "id", row.ID)
		return nil
	}
	ev := factory()
	if err := json.Unmarshal(row.Payload, ev); err != nil {
		slog.Error("durable outbox: falha ao decodificar payload, descartando linha", "eventType", row.EventType, "id", row.ID, "error", err)
		return nil
	}

	if o.publisher != nil {
		return o.publisher.Publish(ctx, ev)
	}

	o.mu.Lock()
	handlers := append([]func(ctx context.Context, ev Event) error(nil), o.handlers[row.EventType]...)
	o.mu.Unlock()

	for _, h := range handlers {
		if err := h(ctx, ev); err != nil {
			return err
		}
	}
	return nil
}
