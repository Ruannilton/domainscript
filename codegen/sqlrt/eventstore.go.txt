package sqlruntime

import (
	"context"
	"database/sql"
	"encoding/json"
	"fmt"
	"time"

	"domainscript/generated/runtime"
)

// EventFactory constrói um novo ponteiro zero-value de um tipo de evento,
// pronto para json.Unmarshal — uma entrada por struct de Event que o módulo
// declara. O registry (mapa EventType() -> EventFactory) é o que permite a
// EventStore.Load decodificar um payload JSON de volta ao tipo Go concreto
// que gravou o evento, sem reflection sobre tipos desconhecidos deste
// pacote. É um ALIAS (não um tipo novo, note o "="): cada módulo gerado já
// declara "var eventRegistry map[string]func() runtime.Event" e o expõe via
// EventRegistry() (codegen/decl_event.go, G1) — um alias faz
// map[string]EventFactory e o mapa devolvido por EventRegistry() serem o
// MESMO tipo Go, então o wiring gerado (cmd/<service>/main.go) passa
// "<pkg>.EventRegistry()" direto a NewEventStore/NewUnitOfWork sem nenhuma
// conversão.
type EventFactory = func() runtime.Event

// EventStore é um runtime.EventStore de verdade sobre database/sql (G1,
// §design 3.11, REQ-26.2): a MESMA interface que
// runtime.NewMemoryEventStore() já satisfaz (codegen/rtsrc/eventstore.go.txt)
// — todo Load<Agg>/UseCase/Query gerado contra runtime.EventStore funciona
// inalterado aqui; só a implementação injetada pelo wiring
// (cmd/<service>/main.go) muda.
type EventStore struct {
	db       *sql.DB
	registry map[string]EventFactory
	dialect  Dialect
}

// NewEventStore embrulha db (já aberto via Open) como um runtime.EventStore,
// decodificando eventos via registry, e garante que a tabela "events" existe
// (idempotente — seguro chamar a cada início de processo). dialect (I7.0,
// REQ-40.1) decide todo o SQL específico de banco — DDL e placeholders —
// consumido daqui e de appendWithinTx/loadWithinQuerier.
func NewEventStore(ctx context.Context, db *sql.DB, registry map[string]EventFactory, dialect Dialect) (*EventStore, error) {
	if dialect == nil {
		return nil, fmt.Errorf("sqlruntime: dialect não pode ser nil")
	}
	s := &EventStore{db: db, registry: registry, dialect: dialect}
	if err := s.ensureSchema(ctx); err != nil {
		return nil, err
	}
	return s, nil
}

// ensureSchema cria a tabela "events" quando ainda não existe: uma linha por
// evento, chaveada por (aggregate_id, sequence) — o mesmo par que
// memoryEventStore usa como posição no stream em memória. "tenant_id"
// (G5, spec §13.1/13.2, row_level) carrega a mesma tag por-stream que
// memoryEventStore mantém em memória (ver tenantStream, rtsrc/
// eventstore.go.txt) — "" para todo evento gravado fora de um contexto de
// tenant, o que preserva o schema/comportamento de antes de G5 byte a byte
// para qualquer módulo sem tenancy. O nome literal da coluna é fixo aqui
// (não configurável pelo "column:" de tenancy{} do mod.ds, ver
// program.Database.TenancyColumn) — este adapter serve UMA tabela de
// domínio por Database, então parametrizar o nome da coluna não muda a
// SEMÂNTICA de isolamento, só a rotulagem; documentado como simplificação
// desta task, não uma limitação de segurança.
// A tabela "outbox" (Marco J, REQ-42.1) também é garantida aqui, ao lado de
// "events": qualquer Database real gerido por um EventStore pode receber um
// Tx.EnqueueOutbox (uow.go.txt) a qualquer momento — criar a tabela junto
// com "events" (em vez de esperar por uma construção separada de
// DurableOutbox, task J2.3) evita uma corrida de "tabela ainda não existe"
// no primeiro EnqueueOutbox e mantém um único dono de schema por Database.
func (s *EventStore) ensureSchema(ctx context.Context) error {
	if _, err := s.db.ExecContext(ctx, s.dialect.CreateEventsTable()); err != nil {
		return fmt.Errorf("sqlruntime: criar tabela events: %w", err)
	}
	if _, err := s.db.ExecContext(ctx, s.dialect.CreateOutboxTable()); err != nil {
		return fmt.Errorf("sqlruntime: criar tabela outbox: %w", err)
	}
	return nil
}

// Append implementa runtime.EventStore: carimba AggregateID/Sequence/
// Timestamp (espelhando memoryEventStore) e persiste cada evento numa
// transação local — atômica para ESTE store, o mesmo contrato "tudo ou nada"
// que memoryEventStore já dá a uma chamada de Append. Atomicidade ENTRE dois
// EventStore é o que UnitOfWork/UnitOfWork2PC (uow.go.txt/twophase.go.txt)
// acrescentam por cima.
func (s *EventStore) Append(ctx context.Context, aggregateID string, events []runtime.Event) error {
	if len(events) == 0 {
		return nil
	}
	tx, err := s.db.BeginTx(ctx, nil)
	if err != nil {
		return fmt.Errorf("sqlruntime: Append: begin: %w", err)
	}
	defer tx.Rollback()
	if err := appendWithinTx(ctx, tx, s.dialect, aggregateID, events); err != nil {
		return err
	}
	if err := tx.Commit(); err != nil {
		return fmt.Errorf("sqlruntime: Append: commit: %w", err)
	}
	return nil
}

// Load implementa runtime.EventStore: um aggregateID desconhecido devolve um
// slice vazio não-nil e nenhum erro (mesmo contrato de memoryEventStore) —
// uma consulta sem linhas já produz exatamente isso naturalmente. Um
// aggregateID que EXISTE mas foi gravado sob um tenant DIFERENTE do de ctx
// (G5, §13.2) devolve (nil, runtime.ErrNotFound) em vez de vazio-e-ok — ver
// a doc de loadWithinQuerier/tenantVisible.
func (s *EventStore) Load(ctx context.Context, aggregateID string) ([]runtime.Event, error) {
	return loadWithinQuerier(ctx, s.db, s.dialect, s.registry, aggregateID)
}

// tenantIDOf/tenantVisible espelham as funções homônimas não-exportadas do
// pacote runtime (rtsrc/eventstore.go.txt) — mesma regra de row_level
// (§13.2/13.3), reimplementada aqui porque são privadas àquele pacote;
// domainscript/generated/runtime já exporta os dois acessores de que
// precisamos (TenantFrom/CrossTenantBypassed) para replicá-la sem acoplar os
// dois pacotes além do necessário.
func tenantIDOf(ctx context.Context) string {
	t, _ := runtime.TenantFrom(ctx)
	return t.ID
}

func tenantVisible(ctx context.Context, tag string) bool {
	if tag == "" || runtime.CrossTenantBypassed(ctx) {
		return true
	}
	t, ok := runtime.TenantFrom(ctx)
	return ok && t.ID == tag
}

// --- helpers compartilhados com uow.go.txt/twophase.go.txt. ---

// appendWithinTx executa o INSERT de cada evento de events dentro de tx já
// aberta — reusada por EventStore.Append (transação local de 1 chamada) e
// por Tx.Append (uow.go.txt/twophase.go.txt: a transação de uma unit of work
// inteira, que só comita/rollbacka quando Run termina). ev.SetMeta roda ANTES
// do encode para que o payload persistido já reflita a metadata carimbada
// (Load volta a carimbar a partir das colunas — autoritativas — na leitura,
// então isso é redundante mas fiel, nunca incorreto).
//
// Tag de tenant (G5, §13.2, row_level): a PRIMEIRA escrita de um stream (next
// == 0 antes de incrementar) carimba tenant_id com o Tenant.ID ativo em ctx
// ("" se nenhum) — escritas SEGUINTES ao MESMO stream reusam a tag JÁ
// gravada (lida abaixo via MAX(tenant_id), constante entre linhas do mesmo
// aggregate_id), nunca a de ctx nessa chamada: um stream não muda de dono no
// meio da vida, mesma regra de memoryEventStore.Append.
func appendWithinTx(ctx context.Context, tx *sql.Tx, dialect Dialect, aggregateID string, events []runtime.Event) error {
	var next int64
	var existingTenant sql.NullString
	nextSeqQuery := fmt.Sprintf(`SELECT COALESCE(MAX(sequence), 0), MAX(tenant_id) FROM events WHERE aggregate_id = %s`, dialect.Placeholder(1))
	row := tx.QueryRowContext(ctx, nextSeqQuery, aggregateID)
	if err := row.Scan(&next, &existingTenant); err != nil {
		return fmt.Errorf("sqlruntime: Append: próxima sequência: %w", err)
	}
	tenantID := existingTenant.String
	if next == 0 {
		tenantID = tenantIDOf(ctx)
	}

	insertQuery := fmt.Sprintf(
		`INSERT INTO events (aggregate_id, sequence, event_type, payload, recorded_at, tenant_id) VALUES (%s, %s, %s, %s, %s, %s)`,
		dialect.Placeholder(1), dialect.Placeholder(2), dialect.Placeholder(3),
		dialect.Placeholder(4), dialect.Placeholder(5), dialect.Placeholder(6),
	)

	now := time.Now().UTC()
	for _, ev := range events {
		next++
		ev.SetMeta(runtime.EventMeta{AggregateID: aggregateID, Sequence: next, Timestamp: now})
		payload, err := json.Marshal(ev)
		if err != nil {
			return fmt.Errorf("sqlruntime: Append: encode %s: %w", ev.EventType(), err)
		}
		if _, err := tx.ExecContext(ctx, insertQuery,
			aggregateID, next, ev.EventType(), string(payload), now, tenantID,
		); err != nil {
			return fmt.Errorf("sqlruntime: Append: insert %s: %w", ev.EventType(), err)
		}
	}
	return nil
}

// querier é o subconjunto de *sql.DB/*sql.Tx que scanEvents/loadWithinQuerier
// precisam — deixa a leitura reusável tanto fora de uma unit of work
// (EventStore.Load, sobre s.db) quanto dentro de uma (Tx.Load, sobre o
// *sql.Tx corrente da unit of work — leitura "read-your-writes" das escritas
// ainda não comitadas do próprio Run).
type querier interface {
	QueryContext(ctx context.Context, query string, args ...any) (*sql.Rows, error)
}

// loadWithinQuerier lê o stream de aggregateID via q (um *sql.DB ou a *sql.Tx
// corrente) e decodifica cada linha usando registry — o corpo de
// EventStore.Load e de Tx.Load (uow.go.txt/twophase.go.txt).
//
// Filtro de tenant (G5, §13.2, row_level): tenant_id é constante entre todas
// as linhas de um mesmo aggregate_id (ver appendWithinTx), então a PRIMEIRA
// linha lida já basta para decidir visibilidade — tenantVisible reprova,
// devolve (nil, runtime.ErrNotFound) IMEDIATAMENTE, sem decodificar o resto
// do stream (mesmo raciocínio do memoryEventStore.Load: nunca vazio-e-ok
// para um stream que EXISTE mas pertence a outro tenant — ver a doc do tipo
// EventStore em rtsrc/eventstore.go.txt).
func loadWithinQuerier(ctx context.Context, q querier, dialect Dialect, registry map[string]EventFactory, aggregateID string) ([]runtime.Event, error) {
	selectQuery := fmt.Sprintf(
		`SELECT sequence, event_type, payload, recorded_at, tenant_id FROM events WHERE aggregate_id = %s ORDER BY sequence`,
		dialect.Placeholder(1),
	)
	rows, err := q.QueryContext(ctx, selectQuery, aggregateID)
	if err != nil {
		return nil, fmt.Errorf("sqlruntime: Load: %w", err)
	}
	defer rows.Close()

	events := make([]runtime.Event, 0)
	tenantChecked := false
	for rows.Next() {
		var seq int64
		var eventType, payload, tenantID string
		var recordedAt time.Time
		if err := rows.Scan(&seq, &eventType, &payload, &recordedAt, &tenantID); err != nil {
			return nil, fmt.Errorf("sqlruntime: Load: scan: %w", err)
		}
		if !tenantChecked {
			tenantChecked = true
			if !tenantVisible(ctx, tenantID) {
				return nil, runtime.ErrNotFound
			}
		}
		factory, ok := registry[eventType]
		if !ok {
			return nil, fmt.Errorf("sqlruntime: Load: tipo de evento %q não registrado (registry incompleto — bug de geração)", eventType)
		}
		ev := factory()
		if err := json.Unmarshal([]byte(payload), ev); err != nil {
			return nil, fmt.Errorf("sqlruntime: Load: decode %s: %w", eventType, err)
		}
		ev.SetMeta(runtime.EventMeta{AggregateID: aggregateID, Sequence: seq, Timestamp: recordedAt})
		events = append(events, ev)
	}
	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("sqlruntime: Load: %w", err)
	}
	return events, nil
}

// enqueueOutboxWithinTx grava cada evento de events na tabela "outbox"
// dentro de tx já aberta (Marco J, REQ-42.1) — reusada por Tx.EnqueueOutbox
// (uow.go.txt/twophase.go.txt), a MESMA *sql.Tx de uma unit of work inteira:
// atomicidade store+outbox por construção (um rollback de Run desfaz as
// duas tabelas, um commit grava as duas). "id" é deixado para o banco
// auto-incrementar (Dialect.CreateOutboxTable) — nunca um UUID: o relay
// (DurableOutbox, task J2.3) entrega por ordem de "id" (FIFO, task J2.2),
// e só um contador monotônico de inserção garante essa ordem.
// delivered_at/attempts ficam nos defaults do schema (NULL/0) — a linha
// nasce "não entregue".
func enqueueOutboxWithinTx(ctx context.Context, tx *sql.Tx, dialect Dialect, events []runtime.Event) error {
	insertQuery := fmt.Sprintf(
		`INSERT INTO outbox (event_type, payload, created_at) VALUES (%s, %s, %s)`,
		dialect.Placeholder(1), dialect.Placeholder(2), dialect.Placeholder(3),
	)
	now := time.Now().UTC()
	for _, ev := range events {
		payload, err := json.Marshal(ev)
		if err != nil {
			return fmt.Errorf("sqlruntime: EnqueueOutbox: encode %s: %w", ev.EventType(), err)
		}
		if _, err := tx.ExecContext(ctx, insertQuery, ev.EventType(), string(payload), now); err != nil {
			return fmt.Errorf("sqlruntime: EnqueueOutbox: insert %s: %w", ev.EventType(), err)
		}
	}
	return nil
}
