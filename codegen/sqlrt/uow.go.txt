package sqlruntime

import (
	"context"
	"database/sql"
	"log/slog"

	"domainscript/generated/runtime"
)

// Tx implementa runtime.Tx sobre um único *sql.Tx — o análogo, sobre
// database/sql, da memoryTx in-memory (runtime/uow.go.txt): as escritas ficam
// dentro de uma transação SQL de verdade, só visíveis a outras conexões após
// o Commit — exatamente o comportamento que o comentário de
// runtime.UnitOfWork já antecipava para este adapter ("buffers writes and
// only makes them visible on commit — that lands with the database/sql
// adapter").
type Tx struct {
	ctx      context.Context
	sqlTx    *sql.Tx
	registry map[string]EventFactory
	dialect  Dialect
	appended []runtime.Event
}

// Append implementa runtime.Tx.
func (t *Tx) Append(aggregateID string, events []runtime.Event) error {
	if err := appendWithinTx(t.ctx, t.sqlTx, t.dialect, aggregateID, events); err != nil {
		return err
	}
	t.appended = append(t.appended, events...)
	return nil
}

// Load implementa runtime.Tx: lê através da MESMA *sql.Tx ainda não comitada,
// então um UseCase que carrega um Aggregate em que ele mesmo (ou um statement
// anterior) já escreveu dentro desta unit of work enxerga suas próprias
// escritas — semântica padrão de transação "read-your-writes".
func (t *Tx) Load(aggregateID string) ([]runtime.Event, error) {
	return loadWithinQuerier(t.ctx, t.sqlTx, t.dialect, t.registry, aggregateID)
}

// EnqueueOutbox implementa runtime.Tx (Marco J, REQ-42.1): grava cada evento
// na tabela "outbox" dentro da MESMA *sql.Tx que Append já usa — commit
// grava as duas tabelas, rollback desfaz as duas. Ver enqueueOutboxWithinTx
// (eventstore.go.txt) para a query/ordenação.
func (t *Tx) EnqueueOutbox(events []runtime.Event) error {
	return enqueueOutboxWithinTx(t.ctx, t.sqlTx, t.dialect, events)
}

// UnitOfWork é o runtime.UnitOfWork de um único banco sobre database/sql (G1,
// §design 3.8): abre uma *sql.Tx de verdade, roda fn e comita no sucesso ou
// rollbacka no erro — o mesmo contrato commit/rollback que memoryUnitOfWork
// já documenta como "landing com o adapter database/sql (Marco G)".
type UnitOfWork struct {
	db        *sql.DB
	registry  map[string]EventFactory
	dialect   Dialect
	publisher runtime.Publisher
}

// NewUnitOfWork cria uma UnitOfWork sobre db (já aberta via Open), decodifica
// eventos via registry — mesma convenção de publisher opcional de
// runtime.NewUnitOfWork (rtsrc/uow.go.txt): quando dado, todo evento
// apensado por um Run bem-sucedido é publicado (dispatcher/canal de saída),
// na mesma ordem de apensação, logo após o commit. dialect (I7.0) é repassado
// a cada Tx que Run cria.
func NewUnitOfWork(db *sql.DB, registry map[string]EventFactory, dialect Dialect, publisher ...runtime.Publisher) *UnitOfWork {
	u := &UnitOfWork{db: db, registry: registry, dialect: dialect}
	if len(publisher) > 0 {
		u.publisher = publisher[0]
	}
	return u
}

// Run implementa runtime.UnitOfWork.
func (u *UnitOfWork) Run(ctx context.Context, fn func(tx runtime.Tx) error) error {
	sqlTx, err := u.db.BeginTx(ctx, nil)
	if err != nil {
		return err
	}
	defer sqlTx.Rollback()
	tx := &Tx{ctx: ctx, sqlTx: sqlTx, registry: u.registry, dialect: u.dialect}
	if err := fn(tx); err != nil {
		return err
	}
	if err := sqlTx.Commit(); err != nil {
		return err
	}
	if u.publisher != nil {
		for _, ev := range tx.appended {
			if pubErr := u.publisher.Publish(ctx, ev); pubErr != nil {
				slog.Error("sqlruntime: uow: falha ao publicar evento após commit", "eventType", ev.EventType(), "error", pubErr)
			}
		}
	}
	return nil
}
