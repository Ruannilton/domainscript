package sqlruntime

import (
	"context"
	"database/sql"
	"fmt"
	"log/slog"
	"sort"

	"domainscript/generated/runtime"
)

// UnitOfWork2PC é o runtime.UnitOfWork2PC (runtime/uow.go.txt) de verdade
// sobre database/sql (G1, REQ-20.5, §design 3.8): coordena um commit em duas
// fases entre todo Database nomeado em Run's dbNames — todos os quais o
// front-end já exigiu que declarassem supportsXA (REQ-5.9) antes deste código
// sequer ser alcançável (o gerador, decl_usecase.go, só emite a chamada a
// este tipo quando esse pré-requisito já foi validado — ver a doc de
// touchedDatabasesFor2PC).
//
// As duas fases, mapeadas sobre o que a API portável de database/sql de fato
// oferece (sem PREPARE TRANSACTION/COMMIT PREPARED específico de driver —
// simplificação documentada, ver o relatório final da task G1):
//
//  1. Prepare: abre uma *sql.Tx por banco e roda fn contra elas. Todo
//     Tx.Append dentro de fn executa seus INSERTs contra a PRÓPRIA *sql.Tx
//     ainda não comitada — qualquer falha (violação de constraint, um Handle
//     de domínio devolvendo um erro de negócio, …) aparece sincronamente como
//     fn devolvendo erro não-nil, o que rollbacka TODO banco envolvido (não
//     só o que falhou) — nenhum banco fica parcialmente aplicado.
//  2. Commit: uma vez que fn devolve nil (todo banco aceitou suas escritas),
//     comita cada *sql.Tx numa ordem determinística (nome do Database,
//     ordenado). Uma falha de commit depois que um commit anterior já
//     sucedeu é um hazard genuíno de "transação em dúvida"/commit parcial —
//     inerente a coordenar commit sobre conexões database/sql independentes
//     sem suporte a XA nativo do driver — logado com destaque como erro
//     crítico e devolvido, em vez de escondido.
type UnitOfWork2PC struct {
	stores    map[string]*EventStore // nome do Database (mod.ds) -> store
	publisher runtime.Publisher
}

// NewUnitOfWork2PC cria um UnitOfWork2PC sobre stores (uma *EventStore por
// Database XA envolvido, construída via NewEventStore/Open pelo wiring
// gerado, cmd/<service>/main.go).
func NewUnitOfWork2PC(stores map[string]*EventStore, publisher ...runtime.Publisher) *UnitOfWork2PC {
	u := &UnitOfWork2PC{stores: stores}
	if len(publisher) > 0 {
		u.publisher = publisher[0]
	}
	return u
}

// Run implementa runtime.UnitOfWork2PC.
func (u *UnitOfWork2PC) Run(ctx context.Context, dbNames []string, fn func(txs map[string]runtime.Tx) error) error {
	sqlTxs := make(map[string]*sql.Tx, len(dbNames))
	innerTxs := make(map[string]*Tx, len(dbNames))
	txs := make(map[string]runtime.Tx, len(dbNames))
	var begunOrder []string

	rollbackAll := func() {
		for i := len(begunOrder) - 1; i >= 0; i-- {
			_ = sqlTxs[begunOrder[i]].Rollback()
		}
	}

	for _, name := range dbNames {
		store, ok := u.stores[name]
		if !ok {
			rollbackAll()
			return fmt.Errorf("sqlruntime: UnitOfWork2PC.Run: nenhum EventStore configurado para o Database %q (bug de geração — G1 só deveria pedir 2PC entre Databases realmente conectados)", name)
		}
		sqlTx, err := store.db.BeginTx(ctx, nil)
		if err != nil {
			rollbackAll()
			return fmt.Errorf("sqlruntime: UnitOfWork2PC.Run: prepare %q: begin: %w", name, err)
		}
		sqlTxs[name] = sqlTx
		begunOrder = append(begunOrder, name)
		tx := &Tx{ctx: ctx, sqlTx: sqlTx, registry: store.registry}
		innerTxs[name] = tx
		txs[name] = tx
	}

	// Fase 1 (prepare): fn executa os Appends de cada banco contra sua
	// própria *sql.Tx ainda não comitada — qualquer erro aqui é um "prepare
	// falhou", tratado com rollback de TODOS os bancos (ver a doc do tipo).
	if err := fn(txs); err != nil {
		rollbackAll()
		return err
	}

	// Fase 2 (commit), em ordem determinística.
	names := make([]string, 0, len(sqlTxs))
	for name := range sqlTxs {
		names = append(names, name)
	}
	sort.Strings(names)

	var committed []string
	for _, name := range names {
		if err := sqlTxs[name].Commit(); err != nil {
			if len(committed) > 0 {
				slog.Error("sqlruntime: UnitOfWork2PC.Run: commit parcial — transação em dúvida entre bancos (limitação documentada de 2PC sobre database/sql sem XA nativo de driver)",
					"bancoComFalha", name, "bancosJaComitados", committed, "error", err)
			}
			return fmt.Errorf("sqlruntime: UnitOfWork2PC.Run: commit %q: %w", name, err)
		}
		committed = append(committed, name)
	}

	if u.publisher != nil {
		for _, name := range names {
			for _, ev := range innerTxs[name].appended {
				if pubErr := u.publisher.Publish(ctx, ev); pubErr != nil {
					slog.Error("sqlruntime: uow2pc: falha ao publicar evento após commit", "eventType", ev.EventType(), "error", pubErr)
				}
			}
		}
	}
	return nil
}
