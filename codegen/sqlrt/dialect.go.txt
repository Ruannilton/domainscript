package sqlruntime

import "fmt"

// Dialect encapsula TUDO que varia por banco na superfície SQL que este
// adapter emite (I7.0, REQ-40.1, §design read-side 3.9a): estilo de
// placeholder, o DDL idempotente das duas únicas tabelas que o gerador
// conhece (events, aqui; a tabela de Collection[T] de REQ-38 entra em I7.1)
// e a forma de paginação. Nenhuma string SQL específica de banco vive fora
// de uma implementação de Dialect — eventstore.go.txt/uow.go.txt/
// twophase.go.txt montam toda query chamando um Dialect, nunca literais "?"
// ou "$N" soltos.
type Dialect interface {
	// Placeholder devolve o marcador do n-ésimo parâmetro (1-based): "?"
	// (sqlite/mysql) ou "$1" (postgres).
	Placeholder(n int) string
	// CreateEventsTable devolve o DDL idempotente da tabela "events" — um
	// stream de eventos por (aggregate_id, sequence).
	CreateEventsTable() string
	// CreateCollectionTable devolve o DDL idempotente da tabela que
	// sustenta um Collection[T] chamado name (REQ-38, adapter de I7.1) —
	// parte do seam desde já para que I7.1 não precise reabrir a interface.
	CreateCollectionTable(name string) string
	// CreateOutboxTable devolve o DDL idempotente da tabela "outbox" (Marco
	// J, REQ-42.1/42.4, §design infra-providers 3.2): o mecanismo de
	// entrega durável at-least-once por trás de DurableOutbox
	// (rtsrc/outbox.go.txt, task J2.3) — cada linha é um evento pendente de
	// entrega, gravado atomicamente com os eventos de negócio (o INSERT
	// entra na MESMA *sql.Tx de Tx.Append — ver Tx.EnqueueOutbox,
	// uow.go.txt). "id" é uma chave técnica AUTO-INCREMENTADA pelo próprio
	// banco (não um UUID): o relay entrega em ordem de "id" (FIFO, task
	// J2.2, "ORDER BY id"), e só um contador monotônico de inserção garante
	// essa ordem — um UUID não ordenaria por chegada.
	CreateOutboxTable() string
	// LimitOffset devolve a cláusula de paginação para limit/offset (LIMIT/
	// OFFSET ANSI serve sqlite e postgres; o método existe para o banco que
	// divergir).
	LimitOffset(limit, offset int) string
	// JSONFieldEq devolve a condição SQL "<extração-json-do-campo> =
	// <placeholder>" usada por Collection[T].whereEqClause
	// (collection.go.txt, REQ-38.1) para cada FieldEq de um WhereEq —
	// TODA a diferença de sintaxe de extração/igualdade de campo JSON vive
	// aqui (REQ-40.1/41.1/41.4): sqlite usa a função
	// json_extract(payload,'$.<campo>'); postgres usa o operador
	// payload->>'<campo>' (extração como TEXTO, não jsonb). placeholder já
	// vem pronto (o resultado de Placeholder(n) — este método não decide
	// o estilo do parâmetro, só a extração do campo). O parâmetro
	// vinculado do lado do chamador é sempre o valor Go stringificado
	// (nunca o valor nativo), porque ambos os operadores de extração
	// devolvem texto — é a mesma régua de segurança de tipo que já
	// restringe quais primitivos entram em WhereEq
	// (codegen/lower/whereeq.go:unsafeWhereEqPrimitives, R7 §design
	// infra-providers 3.1): essa lista é dialeto-agnóstica de propósito,
	// então o conjunto de tipos seguros é idêntico entre SQLiteDialect e
	// PostgresDialect por construção, nunca por auditoria caso a caso.
	JSONFieldEq(field, placeholder string) string
}

// sqliteDialect implementa Dialect para o único driver real que este
// gerador sabe vendorar hoje (modernc.org/sqlite, open_sqlite.go.txt).
type sqliteDialect struct{}

// SQLiteDialect devolve o Dialect usado por todo adapter gerado para um
// Database provider:"sqlite" — o registro único de provider
// (codegen/sql_wiring.go) chama este construtor pelo nome.
func SQLiteDialect() Dialect { return sqliteDialect{} }

func (sqliteDialect) Placeholder(int) string { return "?" }

func (sqliteDialect) CreateEventsTable() string {
	return `CREATE TABLE IF NOT EXISTS events (
	aggregate_id TEXT NOT NULL,
	sequence     INTEGER NOT NULL,
	event_type   TEXT NOT NULL,
	payload      TEXT NOT NULL,
	recorded_at  TIMESTAMP NOT NULL,
	tenant_id    TEXT NOT NULL DEFAULT '',
	PRIMARY KEY (aggregate_id, sequence)
)`
}

func (sqliteDialect) CreateCollectionTable(name string) string {
	return fmt.Sprintf(`CREATE TABLE IF NOT EXISTS %s (
	id      TEXT NOT NULL,
	payload TEXT NOT NULL,
	PRIMARY KEY (id)
)`, name)
}

func (sqliteDialect) CreateOutboxTable() string {
	return `CREATE TABLE IF NOT EXISTS outbox (
	id           INTEGER PRIMARY KEY AUTOINCREMENT,
	event_type   TEXT NOT NULL,
	payload      TEXT NOT NULL,
	created_at   TIMESTAMP NOT NULL,
	delivered_at TIMESTAMP,
	attempts     INTEGER NOT NULL DEFAULT 0
)`
}

func (sqliteDialect) LimitOffset(limit, offset int) string {
	return fmt.Sprintf("LIMIT %d OFFSET %d", limit, offset)
}

func (sqliteDialect) JSONFieldEq(field, placeholder string) string {
	return fmt.Sprintf("json_extract(payload,'$.%s') = %s", field, placeholder)
}
